* Markdown
#+BEGIN_SRC emacs-lisp
(use-package markdown-mode
  :mode
  (("README\\.md\\'" . gfm-mode)
  ("\\.md\\'" . markdown-mode))
  :config
  (add-hook 'markdown-mode-hook 'visual-line-mode))
#+END_SRC
