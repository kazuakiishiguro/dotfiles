 (defun agent-shell-start-in-other-window ()
   "Open the Agent Shell in another window, splitting if needed."
   (interactive)
   (let ((start-buffer (current-buffer)))
     (call-interactively 'agent-shell-new-shell)
     (let ((new-buffer (current-buffer)))
       (unless (eq start-buffer new-buffer)
         (switch-to-buffer start-buffer)
         (let ((target-window (if (one-window-p)
                                  (split-window-horizontally)
                                (next-window))))
           (select-window target-window)
           (switch-to-buffer new-buffer))))))

 (use-package agent-shell
   :ensure t
   :bind (("C-c o" . agent-shell-start-in-other-window))
   :config
   (setq agent-shell-google-authentication
 	(agent-shell-google-make-authentication :login t)))
